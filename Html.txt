    <script>
        document.addEventListener('DOMContentLoaded', loadStudents);
        let currentPhotoBase64 = "";
        let videoStream = null;
        const ADMIN_PASS = "AFL";

        // 1. CLASS LOGIC
        function handleClassChange() {
            const cls = parseInt(document.getElementById('classSelect').value);
            const fee = document.getElementById('feeDisplay');
            const sub = document.getElementById('subjects');
            if(isNaN(cls)) { fee.value=""; sub.value=""; return; }

            let f = 0;
            if(cls<=5) f=300; else if(cls<=8) f=400; else if(cls<=10) f=500; else if(cls==11) f=1000; else f=1500;
            fee.value = "‚Çπ " + f;

            if(cls<=5) sub.value="All General Subjects";
            else if(cls<=10) sub.value="Science, Maths, English";
            else sub.value="Physics, Chemistry, Biology";
        }

        // 2. IMAGE
        function compressImage(src, cb) {
            const img = new Image(); img.src = src;
            img.onload = () => {
                const cvs = document.getElementById('compressCanvas');
                const ctx = cvs.getContext('2d');
                const scale = 200 / img.width;
                cvs.width = 200; cvs.height = img.height * scale;
                ctx.drawImage(img, 0, 0, cvs.width, cvs.height);
                cb(cvs.toDataURL('image/jpeg', 0.7));
            }
        }
        function handleImageUpload(inp) {
            if(inp.files[0]) {
                const r = new FileReader();
                r.onload = (e) => compressImage(e.target.result, (res) => setPhoto(res));
                r.readAsDataURL(inp.files[0]);
            }
        }
        function startCamera() {
            document.getElementById('camera-container').style.display='flex';
            document.getElementById('preview-image').style.display='none';
            navigator.mediaDevices.getUserMedia({video:true}).then(s=>{videoStream=s;document.getElementById('video').srcObject=s;});
        }
        function takeSnapshot() {
            const v = document.getElementById('video');
            const c = document.getElementById('compressCanvas');
            c.width=300; c.height=400;
            c.getContext('2d').drawImage(v,0,0,300,400);
            compressImage(c.toDataURL('image/jpeg',0.8), (res)=>{ setPhoto(res); stopCamera(); });
        }
        function stopCamera() { if(videoStream) videoStream.getTracks().forEach(t=>t.stop()); document.getElementById('camera-container').style.display='none'; }
        function setPhoto(res) { currentPhotoBase64=res; const i=document.getElementById('preview-image'); i.src=res; i.style.display='block'; }

        // 3. DATE FORMATTER
        function formatDate(dateStr) {
            if(!dateStr) return "";
            const [y, m, d] = dateStr.split('-');
            return `${d}-${m}-${y}`;
        }

        // 4. DATABASE
        function registerStudent() {
            const name = document.getElementById('sName').value;
            if(!name) return alert("Name required!");
            let db = JSON.parse(localStorage.getItem('db_safe')) || [];
            let id = "AFL100" + (db.length + 1);
            
            let s = {
                regNo: id,
                sName: name,
                fName: document.getElementById('fName').value,
                mName: document.getElementById('mName').value,
                admDate: document.getElementById('admDate').value,
                classVal: document.getElementById('classSelect').value,
                fee: document.getElementById('feeDisplay').value,
                subjects: document.getElementById('subjects').value,
                photo: currentPhotoBase64
            };

            try {
                db.push(s); localStorage.setItem('db_safe', JSON.stringify(db));
                alert("Registered! ID: " + id);
                resetForm(); loadStudents(); 
                preparePrint([s]); 
            } catch(e) { alert("Storage Full or Error: " + e.message); }
        }

        function editStudent(i) {
            if(prompt("Admin Password:")!==ADMIN_PASS) return alert("Denied");
            let s = JSON.parse(localStorage.getItem('db_safe'))[i];
            document.getElementById('editIndex').value = i;
            document.getElementById('sName').value = s.sName;
            document.getElementById('fName').value = s.fName;
            document.getElementById('mName').value = s.mName;
            document.getElementById('admDate').value = s.admDate;
            document.getElementById('classSelect').value = s.classVal;
            handleClassChange();
            document.getElementById('subjects').value = s.subjects;
            if(s.photo) setPhoto(s.photo);
            document.getElementById('btnSubmit').style.display='none';
            document.getElementById('btnUpdate').style.display='block';
        }

        function updateStudent() {
            let i = document.getElementById('editIndex').value;
            let db = JSON.parse(localStorage.getItem('db_safe'));
            db[i].sName = document.getElementById('sName').value;
            db[i].fName = document.getElementById('fName').value;
            db[i].mName = document.getElementById('mName').value;
            db[i].admDate = document.getElementById('admDate').value;
            db[i].classVal = document.getElementById('classSelect').value;
            db[i].fee = document.getElementById('feeDisplay').value;
            db[i].subjects = document.getElementById('subjects').value;
            if(currentPhotoBase64) db[i].photo = currentPhotoBase64;
            localStorage.setItem('db_safe', JSON.stringify(db));
            alert("Updated!"); resetForm(); loadStudents();
        }

        function deleteStudent(i) {
            if(prompt("Password:")!==ADMIN_PASS) return;
            let db = JSON.parse(localStorage.getItem('db_safe'));
            db.splice(i,1);
            localStorage.setItem('db_safe', JSON.stringify(db));
            loadStudents();
        }

        function resetForm() {
            document.getElementById('regForm').reset();
            currentPhotoBase64="";
            document.getElementById('preview-image').style.display='none';
            document.getElementById('btnSubmit').style.display='block';
            document.getElementById('btnUpdate').style.display='none';
        }

        function loadStudents() {
            let db = JSON.parse(localStorage.getItem('db_safe')) || [];
            let h = "";
            db.forEach((s,i) => {
                h += `<tr><td>${s.regNo}</td><td>${s.sName}</td><td>${s.classVal}</td><td>
                <button class="btn-sm btn-print" onclick="preparePrintByIndex(${i})">Print</button>
                <button class="btn-sm btn-edit" onclick="editStudent(${i})">Edit</button>
                <button class="btn-sm btn-del" onclick="deleteStudent(${i})">Del</button>
                </td></tr>`;
            });
            document.querySelector('#studentTable tbody').innerHTML = h;
        }

        function renderCodes(s, uid) {
            if (typeof QRCode !== 'undefined') {
                try {
                    document.getElementById(`qr_${uid}`).innerHTML="";
                    new QRCode(document.getElementById(`qr_${uid}`), { text: `${s.regNo}|${s.sName}`, width: 60, height: 60 });
                } catch(e) {}
            }
            if (typeof JsBarcode !== 'undefined') {
                try {
                    JsBarcode(`#bc_${uid}`, s.regNo, { format:"CODE128", displayValue:false, height:30, margin:0 });
                } catch(e) {}
            }
        }

        function preparePrintByIndex(i) {
            let s = JSON.parse(localStorage.getItem('db_safe'))[i];
            preparePrint([s]);
        }

        function preparePrint(studentArray) {
            const area = document.getElementById('print-area');
            area.innerHTML = "";
            let queue = [];
            studentArray.forEach((s, idx) => {
                let uid = "p_" + idx + "_" + Date.now();
                area.innerHTML += getCardHTML(s, "STUDENT", uid+"_s") + getCardHTML(s, "OFFICE", uid+"_o") + `<div style="page-break-after:always"></div>`;
                queue.push({s:s, uid:uid});
            });
            document.getElementById('print-overlay').style.display = 'flex';
            setTimeout(() => {
                queue.forEach(q => { renderCodes(q.s, q.uid+"_s"); renderCodes(q.s, q.uid+"_o"); });
            }, 300);
            setTimeout(() => { window.print(); }, 1000);
        }

        function bulkPrint() {
            if(prompt("Password:")!==ADMIN_PASS) return;
            let db = JSON.parse(localStorage.getItem('db_safe')) || [];
            let f = parseInt(document.getElementById('printFrom').value);
            let t = parseInt(document.getElementById('printTo').value);
            let list = [];
            db.forEach((s) => {
                let num = parseInt(s.regNo.replace("AFL100",""));
                if(num >= f && num <= t) list.push(s);
            });
            if(list.length===0) return alert("No students found.");
            preparePrint(list);
        }

        function closeOverlay() {
            document.getElementById('print-overlay').style.display = 'none';
        }
